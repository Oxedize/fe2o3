use oxedize_fe2o3_core::new_enum;
use strum::Display;
#[derive(Clone, Copy, Debug, Display, Eq, PartialEq)]
#[repr(usize)]
pub enum Gender {
    Male,
    Female,
    NonBinary,
    Other,
    PreferNotToSay,
}
new_enum!(Gender;
    Male,
    Female,
    NonBinary,
    Other,
    PreferNotToSay,
);
impl Gender {
    pub fn rand_minority(p: f64) -> Self {
        use ::oxedize_fe2o3_core::rand::Rand;
        let random_val: f64 = Rand::value();
        if random_val < (1.0 - p) {
            if Rand::value::<f64>() < 0.5 {
                Self::Male
            } else {
                Self::Female
            }
        } else {
            let minority_choice = Rand::in_range(0, 3);
            match minority_choice {
                0 => Self::NonBinary,
                1 => Self::Other,
                2 => Self::PreferNotToSay,
                _ => unreachable!(),
            }
        }
    }
}
